#include <stdio.h>
#include <stdlib.h>
#include <veda.h>
#define VEDA(err) check(err, __FILE__, __LINE__)
void check(VEDAresult err, const char* file, const int line) {
if(err != VEDA_SUCCESS) {
const char *name, *str;
vedaGetErrorName (err, &name);
vedaGetErrorString (err, &str);
printf("%s: %s @ %s:%i\n", name, str, file, line);
exit(1);
}
}
int main(int argc, char** argv) {
printf("Hello World from Host!\n");
VEDA(vedaInit(0));
VEDAcontext ctx;
VEDA(vedaDevicePrimaryCtxRetain(&ctx, 0));
VEDA(vedaCtxPushCurrent(ctx));
VEDAmodule mod;
VEDA(vedaModuleLoad(&mod, "libomphello.vso"));
VEDAfunction func;
VEDA(vedaModuleGetFunction(&func, mod, "omp_hello"));
VEDAargs args;
VEDA(vedaArgsCreate(&args));
VEDA(vedaLaunchKernelEx(func, 0, args, 1));
VEDA(vedaCtxSynchronize());
VEDA(vedaExit());
return 0;
}
